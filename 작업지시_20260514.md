## 🗄️ TASK 1: DB 마이그레이션

PostgreSQL에 아래 4개 테이블 추가해줘.
기존 테이블과 충돌 없도록 IF NOT EXISTS 사용.
마이그레이션은 `migrations/003_news_pipeline.sql` 파일로 저장하고,
app.py 시작 시 자동 실행되도록 기존 마이그레이션 패턴 따라서 적용해줘.

```sql
-- 분야별 검색식 (매주 갱신)
CREATE TABLE IF NOT EXISTS news_queries (
  id SERIAL PRIMARY KEY,
  category VARCHAR(30) NOT NULL,
  -- 'ai_infra'|'energy_infra'|'ai_service'|'physical_ai'|'ai_biology'
  query_text TEXT NOT NULL,
  week_label VARCHAR(10) NOT NULL,  -- '2026-W20'
  source VARCHAR(20) DEFAULT 'gemini',  -- 'gemini'|'manual'|'nlm_mcp'
  created_at TIMESTAMP DEFAULT NOW()
);

-- 수집된 원본 뉴스 (full_text는 3.9-E에서 채울 예정)
CREATE TABLE IF NOT EXISTS news_raw (
  id SERIAL PRIMARY KEY,
  category VARCHAR(30),
  title TEXT NOT NULL,
  url TEXT UNIQUE,
  source_name VARCHAR(100),
  pub_date TIMESTAMP,
  summary TEXT,          -- RSS snippet (항상 수집)
  full_text TEXT,        -- 기사 본문 (3.9-E에서 수집, 지금은 NULL)
  week_label VARCHAR(10),
  collected_at TIMESTAMP DEFAULT NOW()
);

-- AI 생성 인사이트 (핵심 테이블)
-- related_news_ids: INT[] 대신 JSON 문자열로 저장 (ORM 호환성)
-- 저장 형태: '[1, 3, 7]' → 읽기: json.loads()
CREATE TABLE IF NOT EXISTS news_insights (
  id SERIAL PRIMARY KEY,
  week_label VARCHAR(10),
  category VARCHAR(30),  -- NULL이면 크로스 분야
  insight_type VARCHAR(30),
  -- 'paradigm_shift'|'hidden_signal'|'value_chain'|'capital_compass'|'landscape_shift'
  title TEXT,
  content TEXT,          -- 마크다운 본문
  related_news_ids TEXT DEFAULT '[]',  -- JSON 문자열로 저장
  ai_source VARCHAR(20) DEFAULT 'gemini',
  status VARCHAR(20) DEFAULT 'draft',
  -- 'draft'|'review'|'published'|'youtube_queue'
  published_at TIMESTAMP,
  created_at TIMESTAMP DEFAULT NOW()
);

-- 파이프라인 실행 로그 (디버깅용)
CREATE TABLE IF NOT EXISTS pipeline_runs (
  id SERIAL PRIMARY KEY,
  week_label VARCHAR(10),
  step INT,              -- 1~5
  status VARCHAR(20),   -- 'running'|'done'|'failed'
  result_summary TEXT,
  started_at TIMESTAMP DEFAULT NOW(),
  finished_at TIMESTAMP
);
```

```python
def parse_related_ids(text_value):
    """DB에서 읽은 JSON 문자열 → Python 리스트"""
    try:
        return json.loads(text_value or '[]')
    except Exception:
        return []

def serialize_related_ids(id_list):
    """Python 리스트 → DB 저장용 JSON 문자열"""
    return json.dumps(id_list if isinstance(id_list, list) else [])
```

### 2-2. 주차 계산 유틸 함수 추가

```python
from datetime import datetime

def get_current_week_label():
    """현재 주차를 '2026-W20' 형식으로 반환"""
    iso = datetime.now().isocalendar()
    return f"{iso[0]}-W{iso[1]:02d}"

def get_recent_week_labels(n=4):
    """최근 n주의 week_label 리스트 반환 (최신순)"""
    from datetime import timedelta
    labels = []
    today = datetime.now()
    for i in range(n):
        d = today - timedelta(weeks=i)
        iso = d.isocalendar()
        labels.append(f"{iso[0]}-W{iso[1]:02d}")
    return labels
```
